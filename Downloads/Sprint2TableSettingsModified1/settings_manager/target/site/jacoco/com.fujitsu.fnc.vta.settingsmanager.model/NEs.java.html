package com.fujitsu.fnc.vta.settingsmanager.model;



import java.util.Date;
import java.util.List;

import org.springframework.data.annotation.Id;
import org.springframework.data.mongodb.core.index.Indexed;
import org.springframework.data.mongodb.core.mapping.Document;

import com.fasterxml.jackson.annotation.JsonIgnore;

import io.micrometer.core.instrument.config.validate.Validated.Secret;
import jakarta.validation.constraints.NotBlank;
import lombok.AllArgsConstructor;
import lombok.Data;
import lombok.NoArgsConstructor;

@Data
@NoArgsConstructor
@AllArgsConstructor
@Document(collection = "NODES")
public class NEs {

    @Id
    private String nodeId;

    @NotBlank(message = Constants.NODE_NOT_EMPTY)
    private String nodeName;

    @Indexed
    private String nodeType;

    @NotBlank(message = Constants.NODE_NOT_EMPTY)
    private String parentNodeGroupId;

    @NotBlank(message = Constants.NODE_NOT_EMPTY)
    private String parentNodeGroupName;

    private String parentNodeId;
    private String parentNodeName;

    @NotBlank(message = Constants.NODE_NOT_EMPTY)
    private String productName;

    @NotBlank(message = Constants.NODE_NOT_EMPTY)
    private String productId;

    @NotBlank(message = Constants.NODE_NOT_EMPTY)
    private String productVersion;

    private String description;
    @Indexed
    private String vendorName;

    @NotBlank(message = Constants.NODE_NOT_EMPTY)
    private String protocol;

    private String snmpSysObjectId;
    private String snmpV1V2cCommunity;
    private String snmpv3SecurityLevel;
    private String snmpv3AuthAlgorithm;
    private String snmpv3PrivAlgorithm;
    @JsonIgnore
    private Secret snmpv3Auth;
    @JsonIgnore
    private Secret snmpv3Priv;
    private String snmpv3EngineId;

    @NotBlank(message = Constants.NODE_NOT_EMPTY)
    private String nodeProfileId;

    @NotBlank(message = Constants.NODE_NOT_EMPTY)
    private String nodeProfileName;

    private String username;
    @JsonIgnore
    private Secret password;

    @Indexed
    private String connectionType;

    private String primaryIpv4Address;

    private String primaryIpv6Address;

    private String primaryPort;
    private String secondaryIpv4Address;
    private String secondaryIpv6Address;
    private String secondaryPort;
    private String externalId;

    private String plugin;
    private String telemetryUsername;
    @JsonIgnore
    private Secret telemetryPassword;
    private String telemetryPort;

    private List<String> domainIdList;
    private List<String> domainNameList;

    private String requestsPerTimePeriod;

    @NotBlank(message = Constants.NODE_NOT_EMPTY)
    @Indexed
    private String adminState;

    private String office;
    private Double latitude;
    private Double longitude;
    private String zone;
    private String area;
    private String region;
    private String facility;
    private String building;
    private String floor;
    private String rack;
    private String location;
    private String memo;
    private String operState;
    private String timezone;
    private Date lastAlarmSyncAt;

    private String primaryGatewayNodeId;
    private String primaryGatewayNodeName;
    private String secondaryGatewayNodeId;
    private String secondaryGatewayNodeName;

    private String nfmPStatsAuxSshUsername;
    @JsonIgnore
    private Secret nfmPStatsAuxSshSecret;

    private String fmPropertyFilter;
    private String fmAdvancedFilter;
    private String pmRegisterLogToFilePropertyFilter;
    private String pmRegisterSasLogToFilePropertyFilter;

    private boolean isDeleted;

    public NEs(String nodeId) {
        this.nodeId = nodeId;
    }
}
