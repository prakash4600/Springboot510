package com.fujitsu.fnc.vta.settingsmanager.model;

import java.time.Instant;
import java.time.LocalDateTime;
import java.util.List;

import org.springframework.data.annotation.Id;
import org.springframework.data.mongodb.core.mapping.Document;

import com.fasterxml.jackson.annotation.JsonInclude;
import com.fasterxml.jackson.annotation.JsonInclude.Include;

import lombok.Data;

@Data
@Document(collection = "entities")
public class Entity {

	@Id
	private String id;

	
	private String name;

	private String description;
	private String memo;

	
	private String nodeName;
	private String nodeId;

	
	private String entityProfileName;
	private String entityProfileId;

	private String parentEntityName;
	private String parentEntityId;
	private String type;

	private String macAddress;
	private String ipv4Address;
	private String ipv6Address;

	private String adminSate;
	private String operState;

	private String mtu;
	private String speed;
	private String pmProfileName;
	private String pmProfileId;
	private String linkId;
	private String linkName;
	private String serviceId;
	private String serviceName;
	private List<String> scopeIdList;
	private List<String> scopeNameList;
	private String externalId;

	@JsonInclude(value = Include.NON_NULL)
	private boolean deleted;
	@JsonInclude(value = Include.NON_NULL)
	private LocalDateTime createdAt;
	@JsonInclude(value = Include.NON_NULL)
	private LocalDateTime updateAt;
	private Instant deletionTimestamp;
	private String snmpOid;
	private String snmpIndex;
	private String snmpType;
	private String snmpTypeId;

}
